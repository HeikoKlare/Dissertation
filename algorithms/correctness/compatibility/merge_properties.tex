\algdef{SE}[DOWHILE]{Do}{doWhile}{\algorithmicdo}[1]{\algorithmicwhile\ #1}%

\begin{algorithmic}[1]

\Procedure{\function{MergeConsistencyVariables}}{$\setted{\tupled{\setted{p}, V_{\setted{p}}}}$}
    \algindentskip
    \State $\mathvariable{stopMerge} \gets \textsc{true}$
    \State $\mathvariable{entries} \gets \sequenced{\tupled{\setted{p}, V_{\setted{p}}}}$ \Comment{Convert input set to sequence}
    \algblockskip

    \Do
        \algindentskip
        \State $\mathvariable{stopMerge} \gets \textsc{true}$
        \State $\mathvariable{results} \gets \setted{}$ % was \tupled{} but does not seem to be necessary
        \algblockskip
        
        \While{$\mathvariable{entries} \neq \sequenced{}$}
            \algindentskip
            \State $\mathvariable{ref} \equalsperdefinition \tupled{\propertyset{P}{\textup{ref}}, V_{\propertyset{P}{\textup{ref}}}} \leftarrow$ $\sequenceindex{entries}{0}$ %\Comment{Reference pair to be compared with others}
            \State $\mathvariable{others} \gets \sequenceindex{entries}{1:}$
            \State $\mathvariable{entries} \gets \sequenced{}$
            \algblockskip

            \For{$\tupled{\propertyset{P}{}, V_{\propertyset{P}{}}} \in \mathvariable{others}$}
                \If{$V_{\propertyset{P}{}} \cap V_{\propertyset{P}{\textup{ref}}} = \varnothing$}
                    \State $\mathvariable{entries} \gets \mathvariable{entries} + \tupled{\propertyset{P}{}, V_{\propertyset{P}{}}}$
                \Else
                    \State $\mathvariable{stopMerge} \gets \textsc{false}$ \label{lst:line:stopMerge}%\Comment{Not all $V_{e_i}$ are pairwise disjoint}
                    \State $\mathvariable{ref} \gets \tupled{\propertyset{P}{} \cup \propertyset{P}{\textup{ref}}, V_{\propertyset{P}{}} \cup V_{\propertyset{P}{\textup{ref}}}}$
                \EndIf
            \EndFor
            \State $\mathvariable{results} \gets \mathvariable{results} \cup \setted{\mathvariable{ref}}$
            \algindentskip
        \EndWhile
        \algblockskip

        \State $\mathvariable{entries} \gets \mathvariable{results}$
        \algindentskip
    \doWhile{$\neg \mathvariable{stopMerge}$}
    \algblockskip

    \State \Return{set($\mathvariable{entries}$)}
    \algindentskip
\EndProcedure

\end{algorithmic}
