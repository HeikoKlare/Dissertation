\begin{algorithmic}[1]
\Procedure{\textsc{Enumerate-Cycles}}{$\text{Dual}\ \mathcal{M}^{*}, \textsc{v} \in V_{\mathcal{M}^{*}}$}
    \State $\{B_1,\dots,B_n\}$ $\leftarrow$ \textsc{Paton-Algorithm}($\mathcal{M}^{*}$)
    \State $Q$ $\leftarrow$ $\{B_1\}$, $R$ $\leftarrow$ $\varnothing$, $R^{*}$ $\leftarrow$ $\varnothing$\\

    \For{$B \in \{B_2, \dots, B_n\}$}
        \For{$T \in Q$}
            \If{$T \cap B \neq \varnothing$}
                \State $R \leftarrow R \cup \{T \oplus B\}$
            \Else
                \State $R^{*} \leftarrow R^{*} \cup \{T \oplus B\}$
            \EndIf
        \EndFor
        \LineComment{Remove non-simple cycles from $R$}
        \For{$U, V \in R$}
            \If{$U \subset V$}
                \State $R \leftarrow R \setminus \{V\}$, $R^{*} \leftarrow R^{*} \cup \{V\}$
            \EndIf
        \EndFor
        \LineComment{New valid cycles are in $R \cup \{B\}$}
        \For{$C \in R \cup \{B\}$}\label{lst:line:newCycles}
            \If{$v \in C$ \textbf{and} \textsc{Replace-Hyperedge}$(v, C)$}
                \State remove $v$ and its incident edges from $\mathcal{M}^{*}$
                \State \textbf{break}
            \EndIf
        \EndFor
        \State $Q \leftarrow Q \cup R \cup R^{*} \cup \{B\}$
        \State $R \leftarrow \varnothing$, $R^{*} \leftarrow \varnothing$
    \EndFor
    %\State \Return{$S$}
\EndProcedure
\end{algorithmic}
