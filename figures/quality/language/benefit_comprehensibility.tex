concept ComponentBasedDesign

commonality Component {
	with uml:Component
	with pcm:Component
	with java:Class

	has name {
		= uml:Component.name
		= pcm:Component.name
		= prefix(java:Class.name,
				"Impl")
		-> firstUpper(
				java:Class.packageName)
	}
}
    \end{lstlisting}
