\node[consistency related element 2, below left=5em and 2em of person.south west, anchor=north west] (relations1) {
$\begin{aligned}
    \consistencyrelation{CR}{PER} =\; &

            \land \mathvariable{e.socsecnumber} = \mathvariable{r.socsecnumber}
        }
\end{aligned}$
};
